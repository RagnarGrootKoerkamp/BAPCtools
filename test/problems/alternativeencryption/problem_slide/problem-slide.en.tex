\newcommand{\maxn}{1000}

\begin{frame}
    \frametitle{\problemtitle}

    \begin{columns}
        \begin{column}[T]{.62\textwidth}
            \begin{itemize}
                \item This is a multi-pass problem.
                    Your program will be run twice for each test case.
                \item
                    In the first pass, encrypt $n \leq \maxn$ strings with lengths up~to~$\maxlen$.
                \item The encrypted text should have the same number of letters as the original text.
                \item For all $i$, the $i$th letter of the encrypted text should differ from the $i$th letter of the original text.
                \item
                    In the second pass, your program will be given the strings as encrypted by
                    the first pass, which it should then decrypt to retrieve the original input.
                \item
                    Your submission may take up to \timelimit{} second for each pass.
                \item A testing tool is provided to help you develop your solution.
            \end{itemize}
