Given $n$, print $n$.

\begin{Input}
    The input consists of:
    \begin{itemize}
        \item One line containing a single integer $0\leq n\leq \maxn$.
    \end{itemize}
\end{Input}

\begin{Output}
    Output a single line containing $n$.
\end{Output}
