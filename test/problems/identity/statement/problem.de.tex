\problemname{}

\newcommand{\maxn}{1000}

Gegeben $n$, gebe $n$ aus.

\begin{Input}
    Die Eingabe besteht aus:
    \begin{itemize}
        \item Einer Zeile mit einem Integer $0\leq n\leq \maxn$.
    \end{itemize}
\end{Input}

\begin{Output}
	Gebe eine Zeile mit $n$ aus.
\end{Output}
