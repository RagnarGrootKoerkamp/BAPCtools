      wd=\paperwidth,%
      sep=0pt,%
      leftskip=4.2ex,%
      rightskip=2.2ex,%
    ]{frametitle}%
    \begin{minipage}{0.4\paperwidth}%
    \rule{0pt}{2.2ex}%
	\ifdefempty{\problemlabel}{%
		\insertframetitle%
	}{%
		\fullproblemtitle %
		\\[0.3em]%
		\tiny%
		Problem Author: \problemauthor%
	}%
    \nolinebreak%
    \end{minipage}%
    \begin{minipage}{0.5\paperwidth}%
      \activitychart%
    \end{minipage}%
    \rule[-2.2ex]{0pt}{2.2ex}%
  \end{beamercolorbox}%
}
